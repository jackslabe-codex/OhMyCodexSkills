```cpp
vector<int> add(vector<int> a, vector<int> b) {
    vector<int> c;
    int carry = 0;
    for (int i = 0; i < a.size() || i < b.size(); i++) {
        if (i < a.size()) carry += a[i];
        if (i < b.size()) carry += b[i];
        _______________________ //在此处填入代码
    }
    if (carry) c.push_back(carry);
    return c;
}
```

A.

```cpp
c.push_back(carry / 10);
carry %= 10;
```

```cpp

carry /= 10;
```

